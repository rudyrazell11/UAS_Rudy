newDf['jenis_usaha'], color ='pink',width = 0.4) plt.xticks(size=12,rotation = "vertical") plt.yticks(size=12) plt.title('Sebaran Jenis Usaha yang ada di Jakarta Barat', size=16) plt.ylabel('jumlah
